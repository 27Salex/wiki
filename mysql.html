¿Qué es MySQL?

MySQL es un sistema de gestión de bases de datos relacional (RDBMS) de código abierto, propiedad de Oracle. Es uno de los más utilizados del mundo, especialmente en aplicaciones web (stack LAMP: Linux, Apache, MySQL, PHP). Usa el lenguaje SQL estándar con extensiones propias.

Características clave:
  • Alto rendimiento y escalabilidad
  • Soporte para múltiples motores de almacenamiento (InnoDB, MyISAM)
  • Replicación maestro-esclavo y maestro-maestro
  • Soporte completo de transacciones ACID (InnoDB)
  • Full-Text Search nativo

AUTO_INCREMENT

Genera valores numéricos únicos automáticamente para la clave primaria.

CREATE TABLE productos (
    id    INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    precio DECIMAL(8,2)
);

-- El id se asigna solo
INSERT INTO productos (nombre, precio) VALUES ('Teclado', 29.99);
INSERT INTO productos (nombre, precio) VALUES ('Ratón', 15.50);

-- Consultar el último ID insertado
SELECT LAST_INSERT_ID();

-- Reiniciar el contador
ALTER TABLE productos AUTO_INCREMENT = 1;

Paginación con LIMIT y OFFSET

MySQL usa LIMIT y OFFSET para paginar resultados. Es crucial para APIs REST y listados web.

-- Sintaxis básica: LIMIT filas OFFSET inicio
SELECT * FROM productos
ORDER BY id
LIMIT 10 OFFSET 0;   -- Página 1 (filas 1-10)

SELECT * FROM productos
ORDER BY id
LIMIT 10 OFFSET 10;  -- Página 2 (filas 11-20)

-- Sintaxis alternativa: LIMIT offset, count
SELECT * FROM productos
ORDER BY id
LIMIT 20, 10;            -- Página 3 (filas 21-30)

-- ✅ Paginación eficiente con cursor (evita OFFSET alto)
SELECT * FROM productos
WHERE id > 100        -- último ID de la página anterior
ORDER BY id
LIMIT 10;
Tip: Para grandes volúmenes de datos, la paginación por cursor (WHERE id > último_id) es mucho más eficiente que OFFSET, ya que este último debe recorrer todas las filas anteriores.

Funciones de Fecha y Hora

Obtener fecha/hora actual

SELECT
    NOW()            AS fecha_hora,      -- 2026-05-18 14:30:00
    CURDATE()        AS solo_fecha,      -- 2026-05-18
    CURTIME()        AS solo_hora,       -- 14:30:00
    UNIX_TIMESTAMP() AS timestamp_unix;   -- 1779367800

Extraer partes de una fecha

SELECT
    YEAR(fecha_alta)    AS anio,
    MONTH(fecha_alta)   AS mes,
    DAY(fecha_alta)     AS dia,
    HOUR(fecha_alta)    AS hora,
    DAYNAME(fecha_alta) AS nombre_dia,   -- 'Monday'
    MONTHNAME(fecha_alta) AS nombre_mes, -- 'May'
    QUARTER(fecha_alta) AS trimestre,
    WEEK(fecha_alta)    AS semana
FROM empleados;

Aritmética de fechas

-- Sumar / restar intervalos
SELECT
    DATE_ADD('2026-01-15', INTERVAL 30 DAY)   AS mas_30_dias,
    DATE_SUB(NOW(), INTERVAL 3 MONTH)      AS hace_3_meses,
    DATE_ADD(NOW(), INTERVAL 1 YEAR)       AS proximo_anio;

-- Diferencia entre fechas
SELECT
    DATEDIFF('2026-12-31', CURDATE())     AS dias_restantes,
    TIMESTAMPDIFF(MONTH, fecha_alta, NOW()) AS meses_trabajados
FROM empleados;

Formatear fechas

SELECT
    DATE_FORMAT(NOW(), '%d/%m/%Y')          AS formato_es,    -- 18/05/2026
    DATE_FORMAT(NOW(), '%W, %d de %M %Y')  AS formato_largo, -- Monday, 18 de May 2026
    DATE_FORMAT(NOW(), '%H:%i:%s')          AS hora_24h;      -- 14:30:00

-- Convertir texto a fecha
SELECT STR_TO_DATE('18-05-2026', '%d-%m-%Y') AS fecha;

Funciones de Texto

SELECT
    CONCAT(nombre, ' ', apellido)       AS nombre_completo,
    CONCAT_WS(', ', ciudad, pais)       AS ubicacion,
    UPPER(nombre)                       AS mayusculas,
    LOWER(email)                        AS minusculas,
    LENGTH(nombre)                      AS longitud_bytes,
    CHAR_LENGTH(nombre)                 AS longitud_chars,
    SUBSTRING(nombre, 1, 3)             AS primeras_3,
    TRIM(nombre)                        AS sin_espacios,
    REPLACE(email, '@old.com', '@new.com') AS email_nuevo,
    LEFT(nombre, 5)                     AS izquierda_5,
    RIGHT(nombre, 5)                    AS derecha_5,
    LPAD(id, 5, '0')                    AS id_con_ceros   -- '00042'
FROM empleados;

Control de Flujo

-- IF / CASE dentro de consultas
SELECT
    nombre,
    salario,
    IF(salario > 40000, 'Alto', 'Normal') AS nivel_if,
    CASE
        WHEN salario >= 60000 THEN 'Senior'
        WHEN salario >= 35000 THEN 'Mid'
        ELSE 'Junior'
    END AS nivel_case,
    IFNULL(telefono, 'No disponible')    AS tel,
    COALESCE(telefono, movil, 'N/A')     AS contacto
FROM empleados;

Full-Text Search

MySQL permite búsquedas de texto completo en columnas CHAR, VARCHAR y TEXT usando índices FULLTEXT.

-- 1. Crear índice FULLTEXT
ALTER TABLE articulos
ADD FULLTEXT INDEX idx_ft_titulo_cuerpo (titulo, cuerpo);

-- 2. Búsqueda en lenguaje natural (por relevancia)
SELECT titulo, MATCH(titulo, cuerpo) AGAINST('base datos relacional') AS relevancia
FROM articulos
WHERE MATCH(titulo, cuerpo) AGAINST('base datos relacional')
ORDER BY relevancia DESC;

-- 3. Búsqueda booleana (operadores +, -, *, "...")
SELECT titulo
FROM articulos
WHERE MATCH(titulo, cuerpo)
      AGAINST('+MySQL -Oracle "stored procedure"' IN BOOLEAN MODE);

ENUM y SET

-- ENUM: un solo valor de la lista
CREATE TABLE tareas (
    id     INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(200),
    estado ENUM('pendiente', 'en_progreso', 'completada', 'cancelada')
           DEFAULT 'pendiente'
);

-- SET: múltiples valores de la lista (separados por coma)
CREATE TABLE usuarios (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    nombre      VARCHAR(100),
    permisos    SET('leer', 'escribir', 'eliminar', 'admin')
);
INSERT INTO usuarios (nombre, permisos)
VALUES ('Ana', 'leer,escribir');

-- Buscar usuarios con permiso específico
SELECT * FROM usuarios
WHERE FIND_IN_SET('admin', permisos) > 0;

Procedimientos Almacenados en MySQL

-- Cambiar delimitador para poder usar ; dentro del procedimiento
DELIMITER //

CREATE PROCEDURE sp_empleados_por_depto(
    IN  p_depto_id  INT,
    OUT p_total     INT
)
BEGIN
    SELECT COUNT(*) INTO p_total
    FROM empleados
    WHERE depto_id = p_depto_id;

    SELECT nombre, salario
    FROM empleados
    WHERE depto_id = p_depto_id
    ORDER BY salario DESC;
END //

DELIMITER ;

-- Llamar al procedimiento
CALL sp_empleados_por_depto(3, @total);
SELECT @total AS total_empleados;
