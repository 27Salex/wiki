DTD

Document Type Definition — Validación de XML

¿Qué es DTD?

DTD (Document Type Definition) es un mecanismo para definir la estructura y las reglas que debe cumplir un documento XML. Permite especificar qué elementos, atributos y entidades son válidos, y en qué orden pueden aparecer.

Características:
  • Define la estructura legal de un documento XML
  • Puede ser interna (dentro del XML) o externa (archivo .dtd)
  • Sintaxis propia (no es XML)
  • No soporta tipos de datos complejos (para eso existe XSD)
  • No soporta namespaces

DTD Interna vs Externa

DTD Interna

Se define dentro del propio documento XML, en la declaración <!DOCTYPE>.

<!-- DTD interna: la definición va entre corchetes [] -->
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE biblioteca [
    <!ELEMENT biblioteca (libro+)>
    <!ELEMENT libro (titulo, autor, anio)>
    <!ELEMENT titulo (#PCDATA)>
    <!ELEMENT autor (#PCDATA)>
    <!ELEMENT anio (#PCDATA)>
    <!ATTLIST libro isbn CDATA #REQUIRED>
]>
<biblioteca>
    <libro isbn="978-0-123456-78-9">
        <titulo>Aprendiendo XML</titulo>
        <autor>Juan Pérez</autor>
        <anio>2025</anio>
    </libro>
</biblioteca>

DTD Externa

Se almacena en un archivo separado .dtd y se referencia desde el XML.

<!-- Archivo: biblioteca.dtd -->
<!ELEMENT biblioteca (libro+)>
<!ELEMENT libro (titulo, autor, anio, editorial?)>
<!ELEMENT titulo (#PCDATA)>
<!ELEMENT autor (#PCDATA)>
<!ELEMENT anio (#PCDATA)>
<!ELEMENT editorial (#PCDATA)>
<!ATTLIST libro isbn CDATA #REQUIRED>

<!-- Archivo: biblioteca.xml -->
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE biblioteca SYSTEM "biblioteca.dtd">
<biblioteca>
    <!-- contenido XML aquí -->
</biblioteca>

Declaración de Elementos

Los elementos se declaran con <!ELEMENT> seguido del nombre y su contenido permitido.

Tipos de contenido

<!-- Texto puro (Parsed Character Data) -->
<!ELEMENT nombre (#PCDATA)>

<!-- Elemento vacío (sin contenido, como <br/>) -->
<!ELEMENT saltoLinea EMPTY>

<!-- Cualquier contenido (elementos y/o texto) -->
<!ELEMENT contenedor ANY>

<!-- Solo elementos hijos (sin texto directo) -->
<!ELEMENT persona (nombre, apellido, email)>

<!-- Contenido mixto (texto + elementos) -->
<!ELEMENT parrafo (#PCDATA | negrita | cursiva)*>

Indicadores de cardinalidad

<!-- Sin indicador: exactamente 1 (obligatorio) -->
<!ELEMENT persona (nombre)>

<!-- ? = 0 o 1 (opcional) -->
<!ELEMENT persona (nombre, telefono?)>

<!-- + = 1 o más (al menos uno) -->
<!ELEMENT biblioteca (libro+)>

<!-- * = 0 o más (cualquier cantidad) -->
<!ELEMENT lista (item*)>

<!-- | = elección (uno u otro) -->
<!ELEMENT pago (tarjeta | efectivo | transferencia)>

<!-- , = secuencia (en ese orden exacto) -->
<!ELEMENT direccion (calle, numero, ciudad, cp)>

<!-- Combinaciones -->
<!ELEMENT pedido (cliente, producto+, descuento?, nota*)>

Declaración de Atributos

Se declaran con <!ATTLIST> especificando el elemento, nombre, tipo y modo.

Tipos de atributos

<!-- CDATA: texto libre -->
<!ATTLIST libro isbn CDATA #REQUIRED>

<!-- Enumeración: solo valores listados -->
<!ATTLIST libro formato (fisico | digital | audio) "fisico">

<!-- ID: identificador único en el documento -->
<!ATTLIST libro id ID #REQUIRED>

<!-- IDREF: referencia a un ID existente -->
<!ATTLIST prestamo libro_ref IDREF #REQUIRED>

<!-- IDREFS: múltiples referencias a IDs -->
<!ATTLIST autor libros_ids IDREFS #IMPLIED>

<!-- NMTOKEN: nombre XML válido (sin espacios) -->
<!ATTLIST producto codigo NMTOKEN #REQUIRED>

Modos de atributos

<!-- #REQUIRED: obligatorio -->
<!ATTLIST libro isbn CDATA #REQUIRED>

<!-- #IMPLIED: opcional, sin valor por defecto -->
<!ATTLIST libro edicion CDATA #IMPLIED>

<!-- Valor por defecto: se usa si no se especifica -->
<!ATTLIST libro idioma CDATA "es">

<!-- #FIXED: valor fijo, no puede cambiarse -->
<!ATTLIST libro tipo CDATA #FIXED "impreso">

<!-- Múltiples atributos del mismo elemento -->
<!ATTLIST empleado
    id        ID     #REQUIRED
    nombre    CDATA  #REQUIRED
    activo    (si | no) "si"
    telefono  CDATA  #IMPLIED
>

Entidades

Las entidades son "variables" reutilizables que se sustituyen al procesar el XML.

Entidades generales (internas)

<!-- Definir entidad -->
<!ENTITY empresa "TechCorp Solutions S.L.">
<!ENTITY copyright "&copy; 2026 TechCorp">

<!-- Usar en el XML -->
<nombre>&empresa;</nombre>
<!-- Se renderiza como: TechCorp Solutions S.L. -->

Entidades externas

<!-- Referencia a un archivo externo -->
<!ENTITY aviso_legal SYSTEM "aviso_legal.xml">

<!-- Entidad pública (recurso conocido) -->
<!ENTITY iso_pais PUBLIC "-//ISO//Countries//EN" "countries.xml">

Entidades de parámetro (solo dentro del DTD)

<!-- Se declaran con % y solo se usan dentro del DTD -->
<!ENTITY % datos_contacto "telefono?, email?, web?">

<!-- Reutilizar en varias declaraciones -->
<!ELEMENT cliente (nombre, %datos_contacto;)>
<!ELEMENT proveedor (razon_social, %datos_contacto;)>

Ejemplo Completo

<!-- tienda.dtd -->
<!ENTITY % info_contacto "telefono?, email">

<!ELEMENT tienda (cliente+, producto+, pedido*)>
<!ATTLIST tienda nombre CDATA #REQUIRED>

<!ELEMENT cliente (nombre, apellido, %info_contacto;)>
<!ATTLIST cliente id ID #REQUIRED>

<!ELEMENT producto (nombre_prod, precio, descripcion?)>
<!ATTLIST producto
    id        ID     #REQUIRED
    categoria (electronica | ropa | hogar) #REQUIRED
>

<!ELEMENT pedido (linea+)>
<!ATTLIST pedido
    fecha       CDATA  #REQUIRED
    cliente_ref IDREF  #REQUIRED
>

<!ELEMENT linea EMPTY>
<!ATTLIST linea
    producto_ref IDREF  #REQUIRED
    cantidad     CDATA  #REQUIRED
>

<!ELEMENT nombre (#PCDATA)>
<!ELEMENT apellido (#PCDATA)>
<!ELEMENT telefono (#PCDATA)>
<!ELEMENT email (#PCDATA)>
<!ELEMENT nombre_prod (#PCDATA)>
<!ELEMENT precio (#PCDATA)>
<!ELEMENT descripcion (#PCDATA)>
<!-- tienda.xml -->
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE tienda SYSTEM "tienda.dtd">
<tienda nombre="MiTienda Online">
    <cliente id="c1">
        <nombre>Ana</nombre>
        <apellido>García</apellido>
        <email>ana@correo.com</email>
    </cliente>
    <producto id="p1" categoria="electronica">
        <nombre_prod>Teclado mecánico</nombre_prod>
        <precio>89.99</precio>
        <descripcion>Teclado con switches Cherry MX</descripcion>
    </producto>
    <pedido fecha="2026-05-18" cliente_ref="c1">
        <linea producto_ref="p1" cantidad="2"/>
    </pedido>
</tienda>
