T-SQL — Stored Procedures, Triggers, Funciones

¿Qué es T-SQL?

Transact-SQL (T-SQL) es la extensión propietaria del lenguaje SQL utilizada por Microsoft SQL Server y Azure SQL Database. Añade capacidades procedimentales al SQL estándar: variables, control de flujo, manejo de errores, y más.

Diferencias clave con SQL estándar:
  • TOP en lugar de LIMIT para limitar filas
  • OFFSET...FETCH para paginación (SQL Server 2012+)
  • Variables con prefijo @
  • Tablas temporales con # y ##
  • Manejo de errores con TRY...CATCH

Variables y Tipos

-- Declarar y asignar variables
DECLARE @nombre   NVARCHAR(100) = N'Carlos López';
DECLARE @salario  DECIMAL(10,2);
DECLARE @fecha    DATE = GETDATE();

-- Asignar con SET o SELECT
SET @salario = 45000.00;

SELECT @salario = salario
FROM empleados
WHERE nombre = @nombre;

-- Imprimir valor
PRINT 'Salario: ' + CAST(@salario AS NVARCHAR(20));

Paginación en MSSQL

-- TOP: limitar resultados (sin OFFSET)
SELECT TOP 10 nombre, salario
FROM empleados
ORDER BY salario DESC;

-- OFFSET...FETCH: paginación real (SQL Server 2012+)
SELECT nombre, salario
FROM empleados
ORDER BY id
OFFSET 20 ROWS          -- Saltar las primeras 20 filas
FETCH NEXT 10 ROWS ONLY; -- Tomar las siguientes 10

-- Paginación dinámica con variables
DECLARE @pagina INT = 3;
DECLARE @filas  INT = 10;

SELECT nombre, salario
FROM empleados
ORDER BY id
OFFSET (@pagina - 1) * @filas ROWS
FETCH NEXT @filas ROWS ONLY;

Control de Flujo

-- IF...ELSE
DECLARE @total INT;
SELECT @total = COUNT(*) FROM empleados;

IF @total > 100
BEGIN
    PRINT 'Empresa grande: ' + CAST(@total AS VARCHAR);
END
ELSE IF @total > 20
BEGIN
    PRINT 'Empresa mediana';
END
ELSE
BEGIN
    PRINT 'Empresa pequeña';
END;

-- WHILE
DECLARE @i INT = 1;
WHILE @i <= 10
BEGIN
    PRINT 'Iteración: ' + CAST(@i AS VARCHAR);
    SET @i = @i + 1;
END;

Stored Procedures

Los procedimientos almacenados son bloques de T-SQL compilados y almacenados en el servidor. Mejoran el rendimiento, la reutilización y la seguridad.

Procedimiento básico con parámetros

CREATE OR ALTER PROCEDURE sp_BuscarEmpleados
    @departamento  NVARCHAR(50),
    @salario_min   DECIMAL(10,2) = 0,    -- valor por defecto
    @total         INT OUTPUT              -- parámetro de salida
AS
BEGIN
    SET NOCOUNT ON; -- No enviar mensajes de conteo

    SELECT
        e.nombre,
        e.salario,
        d.nombre AS departamento
    FROM empleados e
    INNER JOIN departamentos d ON e.depto_id = d.id
    WHERE d.nombre = @departamento
      AND e.salario >= @salario_min
    ORDER BY e.salario DESC;

    -- Asignar total al parámetro OUTPUT
    SET @total = @@ROWCOUNT;
END;

-- Ejecutar el procedimiento
DECLARE @resultado INT;
EXEC sp_BuscarEmpleados
    @departamento = N'IT',
    @salario_min  = 30000,
    @total        = @resultado OUTPUT;

PRINT 'Encontrados: ' + CAST(@resultado AS VARCHAR);

Procedimiento con Lógica de Control (IF/ELSE) y PRINT

Este procedimiento muestra la sintaxis de paso de parámetros de entrada y salida, el control de flujo con condicionales IF/ELSE y el uso de PRINT para depuración y mensajería en consola.

CREATE OR ALTER PROCEDURE sp_ProcesarBonoEmpleado
    @depto_id     INT,
    @bono_base    DECIMAL(10,2),
    @filas_afect  INT OUTPUT  -- Parámetro de salida
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @mensaje NVARCHAR(250);
    
    -- Lógica de control IF/ELSE: verificar si el departamento existe
    IF EXISTS (
        SELECT 1 FROM departamentos WHERE id = @depto_id
    )
    BEGIN
        -- Modificar registros basados en el parámetro de entrada
        UPDATE empleados
        SET salario = salario + @bono_base
        WHERE depto_id = @depto_id;
        
        -- Asignar el número de filas afectadas a la variable de salida
        SET @filas_afect = @@ROWCOUNT;
        
        -- Concatenar y mostrar mensaje de éxito en consola usando PRINT
        SET @mensaje = N'Bono procesado con éxito. Empleados afectados: ' 
                     + CAST(@filas_afect AS NVARCHAR(10));
        PRINT @mensaje;
    END
    ELSE
    BEGIN
        SET @filas_afect = 0;
        PRINT N'ERROR: El ID de departamento especificado no existe.';
    END
END;

Procedimiento con manejo de errores

CREATE PROCEDURE sp_TransferirSaldo
    @cuenta_origen   INT,
    @cuenta_destino  INT,
    @monto           DECIMAL(10,2)
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRY
        BEGIN TRANSACTION;

            UPDATE cuentas SET saldo = saldo - @monto
            WHERE id = @cuenta_origen;

            IF (SELECT saldo FROM cuentas WHERE id = @cuenta_origen) < 0
                THROW 50001, 'Saldo insuficiente', 1;

            UPDATE cuentas SET saldo = saldo + @monto
            WHERE id = @cuenta_destino;

        COMMIT TRANSACTION;
        PRINT 'Transferencia exitosa';
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        PRINT 'ERROR: ' + ERROR_MESSAGE();
        PRINT 'Línea: ' + CAST(ERROR_LINE() AS VARCHAR);
    END CATCH;
END;

Funciones Definidas por el Usuario

Las funciones de T-SQL permiten encapsular lógica reutilizable. Tienen restricciones estrictas (no pueden realizar operaciones de escritura en la base de datos). Aspectos sintácticos clave:

  • RETURNS: Especifica obligatoriamente el tipo de dato que devolverá la función (escalar o una estructura de tabla).
  • DECLARE: Permite declarar variables locales de forma explícita especificando su tipo de datos.
  • SET: Se utiliza para realizar asignaciones directas de valores a variables locales previamente declaradas.
  • RETURN: Finaliza la ejecución de la función y devuelve el valor calculado al entorno que lo invocó.

Función Escalar

-- Devuelve un único valor
CREATE FUNCTION fn_CalcularImpuesto(
    @salario DECIMAL(10,2)
)
RETURNS DECIMAL(10,2)
AS
BEGIN
    DECLARE @impuesto DECIMAL(10,2);
    SET @impuesto = CASE
        WHEN @salario > 60000 THEN @salario * 0.30
        WHEN @salario > 30000 THEN @salario * 0.20
        ELSE @salario * 0.10
    END;
    RETURN @impuesto;
END;

-- Uso
SELECT nombre, salario, dbo.fn_CalcularImpuesto(salario) AS impuesto
FROM empleados;

Función con Valor de Tabla (TVF)

-- Devuelve una tabla como resultado
CREATE FUNCTION fn_EmpleadosPorDepto(@depto_id INT)
RETURNS TABLE
AS
RETURN (
    SELECT nombre, salario, fecha_alta
    FROM empleados
    WHERE depto_id = @depto_id
);

-- Uso (como una tabla en FROM)
SELECT * FROM fn_EmpleadosPorDepto(3)
WHERE salario > 30000;

Triggers (Disparadores)

Un trigger es un bloque de código que se ejecuta automáticamente cuando ocurre un evento (INSERT, UPDATE, DELETE) en una tabla. SQL Server usa las pseudo-tablas inserted y deleted.

Trigger AFTER INSERT — Auditoría

-- Tabla de auditoría
CREATE TABLE auditoria_empleados (
    id         INT IDENTITY(1,1) PRIMARY KEY,
    accion     VARCHAR(10),
    emp_id     INT,
    nombre     NVARCHAR(100),
    fecha      DATETIME DEFAULT GETDATE(),
    usuario    NVARCHAR(100) DEFAULT SUSER_SNAME()
);

-- Trigger que registra cada inserción
CREATE TRIGGER trg_Empleados_Insert
ON empleados
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;
    INSERT INTO auditoria_empleados (accion, emp_id, nombre)
    SELECT 'INSERT', id, nombre
    FROM inserted;  -- pseudo-tabla con las filas nuevas
END;

Trigger AFTER UPDATE — Registro de cambios

CREATE TRIGGER trg_Empleados_Update
ON empleados
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;
    -- Solo registrar si cambió el salario
    IF UPDATE(salario)
    BEGIN
        INSERT INTO auditoria_empleados (accion, emp_id, nombre)
        SELECT 'UPDATE', i.id, i.nombre
        FROM inserted i
        INNER JOIN deleted d ON i.id = d.id
        WHERE i.salario <> d.salario; -- realmente cambió
    END;
END;

Trigger AFTER DELETE — Auditoría de Eliminaciones

El trigger AFTER DELETE se ejecuta después de que los datos se han eliminado físicamente de la tabla. Permite capturar las filas borradas a través de la tabla lógica deleted.

CREATE TRIGGER trg_Empleados_Delete
ON empleados
AFTER DELETE
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Intercepta el evento de borrado y registra los datos eliminados
    INSERT INTO auditoria_empleados (accion, emp_id, nombre)
    SELECT 'DELETE', id, nombre
    FROM deleted;  -- pseudo-tabla con las filas eliminadas (DELETED)
    
    PRINT 'Eliminación registrada en auditoría para ' 
        + CAST(@@ROWCOUNT AS VARCHAR) 
        + ' empleado(s).';
END;

Trigger INSTEAD OF — Prevenir eliminaciones

-- En vez de eliminar, marca como inactivo
CREATE TRIGGER trg_Empleados_SoftDelete
ON empleados
INSTEAD OF DELETE
AS
BEGIN
    SET NOCOUNT ON;
    UPDATE e
    SET e.activo = 0,
        e.fecha_baja = GETDATE()
    FROM empleados e
    INNER JOIN deleted d ON e.id = d.id;

    PRINT 'Soft delete aplicado a '
        + CAST(@@ROWCOUNT AS VARCHAR)
        + ' empleado(s)';
END;

Tablas Temporales y Variables de Tabla

-- Tabla temporal LOCAL (visible solo en la sesión actual)
CREATE TABLE #TempVentas (
    producto  VARCHAR(100),
    total     DECIMAL(10,2)
);
INSERT INTO #TempVentas
SELECT producto, SUM(monto) FROM ventas GROUP BY producto;

-- Tabla temporal GLOBAL (visible para todas las sesiones)
CREATE TABLE ##GlobalTemp (id INT, dato VARCHAR(50));

-- Variable de tabla (solo vive en el batch actual)
DECLARE @resumen TABLE (
    departamento VARCHAR(50),
    total        INT
);
INSERT INTO @resumen
SELECT d.nombre, COUNT(e.id)
FROM departamentos d
LEFT JOIN empleados e ON d.id = e.depto_id
GROUP BY d.nombre;

SELECT * FROM @resumen;

SQL Dinámico

Permite construir y ejecutar sentencias SQL en tiempo de ejecución. Usar con sp_executesql para prevenir inyección SQL.

DECLARE @tabla    NVARCHAR(128) = N'empleados';
DECLARE @columna  NVARCHAR(128) = N'nombre';
DECLARE @filtro   NVARCHAR(100) = N'Ana%';
DECLARE @sql      NVARCHAR(MAX);

-- ✅ SEGURO: sp_executesql con parámetros
SET @sql = N'SELECT * FROM ' + QUOTENAME(@tabla)
         + N' WHERE ' + QUOTENAME(@columna) + N' LIKE @p_filtro';

EXEC sp_executesql @sql,
    N'@p_filtro NVARCHAR(100)',
    @p_filtro = @filtro;
